def db
  @db ||=
    Sequel.connect(
      adapter: 'postgres',
      host: 'localhost',
      database: ENV['DATABASE_NAME'],
      user: ENV['DATABASE_USER'],
      password: ENV['DATABASE_PASSWORD']
    )
end
